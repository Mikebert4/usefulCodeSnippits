/**
* converts a week number into a date range
*
*/
function getDateRangeOfWeek(weekNo){
	var d1 = new Date();
	numOfdaysPastSinceLastMonday = eval(d1.getDay()- 1);
	d1.setDate(d1.getDate() - numOfdaysPastSinceLastMonday);
	var weekNoToday = d1.getWeek();
	var weeksInTheFuture = eval( weekNo - weekNoToday );
	d1.setDate(d1.getDate() + eval( 7 * weeksInTheFuture ));
	var rangeIsFrom =  d1.toLocaleDateString();
	d1.setDate(d1.getDate() + 6);		
	var rangeIsTo = d1.toLocaleDateString() ;
	return rangeIsFrom + " to "+rangeIsTo;
};


/**
* converts a week numer (1-52) plus a day number (1-7) into a date
*
*/
function getBookedDay(weekNo,dayNo){
	var d = new Date();
	numOfdaysPastSinceLastMonday = eval(d.getDay()- 1);
	d.setDate(d.getDate() - numOfdaysPastSinceLastMonday);
	var weekNoToday = d.getWeek();
	var weeksInTheFuture = eval( weekNo - weekNoToday );
	d.setDate(d.getDate() + eval( 7 * weeksInTheFuture ));
	d.setDate(d.getDate() + dayNo);		
	var dayBooked = d.toLocaleDateString() ;
	return dayBooked;
};
